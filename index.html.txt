  <script>
    const correctIndex = 0;

    const checkBtn = document.getElementById("check-btn");
    const resultBox = document.getElementById("result");

    checkBtn.addEventListener("click", () => {
      const selected = document.querySelector('input[name="answer"]:checked');

      if (!selected) {
        resultBox.textContent = "Seleziona un'opzione prima di verificare.";
        resultBox.className = "result ko";
        resultBox.style.display = "block";
        return;
      }

      const value = parseInt(selected.value, 10);

      if (value === correctIndex) {
        resultBox.innerHTML =
          "<strong>Corretto!</strong> Le scatole di cartone delle uova sono considerate un materiale <strong>fonoassorbente</strong> (pur con prestazioni limitate rispetto a materiali professionali).";
        resultBox.className = "result ok";
      } else {
        resultBox.innerHTML =
          "<strong>Risposta errata.</strong><br/>In acustica, le scatole di cartone delle uova vengono considerate principalmente un materiale <strong>fonoassorbente</strong>.";
        resultBox.className = "result ko";
      }

      resultBox.style.display = "block";
    });
  </script>
